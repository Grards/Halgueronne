Users:
    id (INT - 11)
    login (VAR - 75) // unique
    password (VAR - 255)
    mail (VAR - 75) // unique
    avatar (VAR - 255) NULL
    role (VAR - 75)
    posts (INT - 11)
    slug (VAR - 255) // unique

messages:
    id (INT - 11)
    sujet (VAR - 100)
    contenu (TEXT)
    date_envoi
    statut (VAR - 10) // Non-lu, lu, répondu, brouillon, envoyé, archivé.
    auteur (INT - 11) -> ManyToOne // Plusieurs messages par auteur.
    destinataire (TEXT) -> OneToMany // Chaque message peut avoir plusieurs destinataires.
                           Géré en TEXT car on recevra une liste avec des séparateurs ',' entre chaque id utilisateur et un statut pour chacun d'eux, genre :
                           d:{id}(destinataire principal),cv:{id}(copie_visible),ci:{id}(copie invisible)

badges:
    id (INT - 11)
    titre (VAR - 75)
    description (TEXT)
    image (VAR - 255)
    slug (VAR - 255)

badges_utilisateurs:
    id (INT - 11)
    date_obtention (DATETIME)
    badge (INT - 11) -> OneToOne // Une seule obtention de badge par utilisateur
    utilisateur (INT - 11) -> OneToOne // Un seul utilisateur par obtention de badge

personnages:
    id (INT - 11)
    nom (VAR - 75)
    prenom (VAR - 75)
    sexe (VAR - 1)
    portrait (VAR - 255)
    naissance_jour (SMALLINT) NULL -> La naissance et la mort sont découpées en champs SMALLINT car il s'agit d'un calendrier spécifique au jeu et non du calendrier terrien.
    naissance_mois (SMALLINT) NULL -> Les valeurs peuvent être nulles car on peut ne pas connaitre la date de naissance exacte d'un personnage et qu'il n'est pas forcément déjà mort.
    naissance_annee (SMALLINT) NULL
    mort_jour (SMALLINT) NULL -> Si au moins un champ de mort est rempli, alors le personnage ne sera plus disponible en jeu.
    mort_mois (SMALLINT) NULL
    mort_annee (SMALLINT) NULL
    biographie (TEXT)
    slug (VAR - 255)
    utilisateur (INT - 11) -> ManyToOne // Plusieurs personnages pour un utilisateur. (Mais via leur rang, les MJ's pourront avoir accès à TOUS les peronnages + édition, etc...)

forums_posts:
    id (INT - 11)
    titre (VAR - 75)
    message (LONGTEXT)
    ordre (SMALLINT)
    date_creation (DATETIME)
    derniere_edition (DATETIME) NULL // Si NULL, pas d'affichage d'édition effectuée sur le poste.

    slug (VAR - 255)
